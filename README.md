// Area-of-rectangle
#include<iostream>
using namespace std;
int main()
{
   int length, breadth, area;
  
   cout << "\n enter the length of rectangle : ";
   cin>>length;

   cout << "\n enter the breadth of rectangle : ";
   cin>>breadth;

   area = length * breadth;
   cout << "\n area of Rectangle : " << area;
 
   return (0);
}
